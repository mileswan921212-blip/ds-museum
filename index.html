Queues

FIFO (First-In, First-Out) operations with visual demo.

Queue JavaScript Code

let queue = [];

function enqueue() {
    const newItem = "Q" + (queue.length + 1);
    queue.push(newItem);
    renderQueue();
}

function dequeue() {
    if (queue.length > 0) {
        queue.shift();
        renderQueue();
    }
}

function renderQueue() {
    const box = document.getElementById("queueBox");
    box.innerHTML = "";

    queue.forEach((item) => {
        const div = document.createElement("div");
        div.className = "queue-item";
        div.innerText = item;
        box.appendChild(div);
    });
}

Linked Lists

Build and visualize a linked list step by step.

Linked List JavaScript Code

let linkedList = [];

function addNode() {
    const value = "N" + (linkedList.length + 1);
    linkedList.push(value);
    renderLinkedList();
}

function removeNode() {
    if (linkedList.length > 0) {
        linkedList.pop();
        renderLinkedList();
    }
}

function renderLinkedList() {
    const container = document.getElementById("llContainer");
    container.innerHTML = "";

    linkedList.forEach((value, index) => {
